import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Load dataset (synthetic example)
data = {
    "employee_id": ["EMP001","EMP002","EMP003","EMP004","EMP005"],
    "department": ["Sales","Sales","Marketing","Operations","IT"],
    "region": ["Europe","Latin America","North America","Latin America","Asia Pacific"],
    "performance_score": [73.76,75.19,80.96,84.70,89.02],
    "years_experience": [9,10,7,9,13],
    "satisfaction_rating": [4.9,4.6,4.4,3.3,3.3]
}

df = pd.DataFrame(data)

# Frequency count for IT department
it_count = (df["department"] == "IT").sum()
print("IT Department Count:", it_count)

# Histogram of departments (example counts)
df["department"].value_counts().plot(kind="bar")
plt.title("Department Distribution")
plt.xlabel("Department")
plt.ylabel("Count")
plt.show()
